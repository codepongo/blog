<h1>List</h1>

<pre><code>* 1
   + 1-1
      - 1-1-1
        - 1-1-2
   + 1-2
* 2
* 3
  1. 3-1
  2. 3-2</code></pre>

<ul>
<li>1
<ul>
<li>1-1</li>
<li>1-1-1</li>
<li>1-1-2</li>
<li>1-2</li>
</ul></li>
<li>2</li>
<li>3
<ol>
<li>3-1</li>
<li>3-2</li>
</ol></li>
</ul>
